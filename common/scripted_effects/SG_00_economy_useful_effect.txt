economy_progress_reset = {
	custom_effect_tooltip = economy_progress_reset_tt
	hidden_effect = {
		set_variable = { economy_progress = global.economy_progress_ini }
	}
}

#				  #
# LEVEL UP / DOWN #
#				  #

economy_level_up = {
	custom_effect_tooltip = economy_level_up_tt

	add_to_variable = { economy_level = 1 }

	economy_progress_reset = yes
	economy_mod_calc = yes

}
economy_level_down = {
	custom_effect_tooltip = economy_level_down_tt

	add_to_variable = { economy_level = -1 }

	economy_progress_reset = yes
	economy_mod_calc = yes

}

#				  #
# IDEA UP / DOWN  #
#				  #

economic_investment_up = {
	if = {
		limit = { has_idea = personal_consignment }
		swap_ideas = {
			remove_idea = personal_consignment
			add_idea = private_economy
		}
	}
	else_if = {
		limit = { has_idea = private_economy }
		swap_ideas = {
			remove_idea = private_economy
			add_idea = reduced_welfare
		}
	}
	else_if = {
		limit = { has_idea = reduced_welfare }
		swap_ideas = {
			remove_idea = reduced_welfare
			add_idea = early_mobilization
		}
	}
	else_if = {
		limit = { has_idea = early_mobilization }
		swap_ideas = {
			remove_idea = early_mobilization
			add_idea = military_state
		}
	}
	else = {
		custom_effect_tooltip = economic_law_limit_tt
	}
}
