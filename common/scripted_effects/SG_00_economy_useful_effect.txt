	add_to_variable = { economy_level = 1 }

economic_investment_up = {
	if = {
		limit = { has_idea = personal_consignment_0 }
		swap_ideas = {
			remove_idea = personal_consignment_0
			add_idea = private_economy_1
		}
	}
	else_if = {
		limit = { has_idea = private_economy_1 }
		swap_ideas = {
			remove_idea = private_economy_1
			add_idea = reduced_welfare_2
		}
	}
	else_if = {
		limit = { has_idea = reduced_welfare_2 }
		swap_ideas = {
			remove_idea = reduced_welfare_2
			add_idea = early_mobilization_3
		}
	}
	else_if = {
		limit = { has_idea = early_mobilization_3 }
		swap_ideas = {
			remove_idea = early_mobilization_3
			add_idea = military_state_4
		}
	}
	else = {
		custom_effect_tooltip = economic_law_limit_tt
	}
}
