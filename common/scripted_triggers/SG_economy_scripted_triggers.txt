has_surplus_economy = {
	custom_trigger_tooltip = {
		tooltip = SG_is_surplus
		check_variable = { economy_index > 0 }
	}
}
has_deficit_economy = {
	custom_trigger_tooltip = {
		tooltip = SG_is_deficit
		check_variable = { economy_index < 0 }
	}
}

available_personal_consignment_0 = {
	always = yes
}
available_private_economy_1 = {
	OR = {
		has_country_flag = available_private_economy_1
		always = yes
	}
}
available_reduced_welfare_2 = {
	OR = {
		has_country_flag = available_reduced_welfare_2
		has_war_support > 0.25
	}
}
available_early_mobilization_3 = {
	OR = {
		has_country_flag = available_early_mobilization_3
		has_war_support > 0.5
	}
}
available_military_state_4 = {
	OR = {
		has_country_flag = available_military_state_4
		AND = {
			has_war_support > 0.75
			has_war = yes
		}
	}
}

available_open_trade_0 = {

	}

available_volunteer_only_1 = {

		}
