has_surplus_economy = {
	custom_trigger_tooltip = {
		tooltip = SG_is_surplus
		check_variable = { economy_index > 0 }
	}
}
has_deficit_economy = {
	custom_trigger_tooltip = {
		tooltip = SG_is_deficit
		check_variable = { economy_index < 0 }
	}
}

available_investment_up = {
	custom_trigger_tooltip = {
		tooltip = SG_available_investment_up

		if = {
			limit = { check_variable = { investment_law_range = 0 } }
			NOT = { has_idea = personal_consignment_0 }
		}
		else_if = {
			limit = { check_variable = { investment_law_range = 1 } }
			NOT = { has_idea = private_economy_1 }
		}
		else_if = {
			limit = { check_variable = { investment_law_range = 2 } }
			NOT = { has_idea = reduced_welfare_2 }
		}
		else_if = {
			limit = { check_variable = { investment_law_range = 3 } }
			NOT = { has_idea = early_mobilization_3 }
		}
		else = {
			NOT = { has_idea = military_state_4 }
		}
	}
}
